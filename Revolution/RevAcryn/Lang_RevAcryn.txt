    //Код для смены акрина на сборочном дроиде, а также сбросе прицела у всех орудий, если дроид установлен не на станции
    RevAcrynAssemblyDroidActCode ^{
        00=[t_OnItemEquip,t_OnItemDeEquip,t_OnItemDestroy,t_OnStep|0]
        01=if(ScriptItemActionType(t_OnStep, 0))
        02 ^{
            01=dword item = CurItem();
            02=dword ship = ScriptItemActShip();
            03=if(!ship) exit;
            04=if(!item) exit;
            05=if(!ItemIsInUse(item)) exit;
            06=if(ItemIsBroken(item)) exit;
            07=if(ObjectType(ship) != 4)
            08 ^{
                01=if(ShipType(ship) == 'DefPlatform') exit;
                02=for(int i = 1; i <= ShipCntWeapon(ship); i = i + 1)
                03 ^{
                    01=WeaponTarget(ShipWeapon(ship, i), 0);
                }
            }
        }
        03=else if(ScriptItemActionType(t_OnItemEquip))
        04 ^{
            01=item = CurItem();
            02=dword star = ShipStar(Player());
            03=int check = 0;
            //Начинаем искать, где же находится наш дроид
            //А всё потому, что ебучие события t_OnItemEquip и t_OnItemDeEquip НЕ ВОЗВРАЩАЮТ, СУКА, ЭКИПИРОВАВШИЙ ПРЕДМЕТ КОРАБЛЬ!
            //Сперва чекаем игрока, как наиболее вероятного кандидата
            04=for(int j = 0; j < ShipItems(Player()); j = j + 1)
            05 ^{
                01=dword temp_item = ShipItems(Player(), j);
                02=if(temp_item == item)
                03 ^{
                    01=check = 1;
                    02=break;
                }
            }
            //Затем станции
            06=if(!check)
            07 ^{
                01=for(j = 0; j < StarRuins(star); j = j + 1)
                02 ^{
                    01=ship = StarRuins(star, j);
                    02=for(int k = 0; k < ShipItems(ship); k = k + 1)
                    03 ^{
                        01=temp_item = ShipItems(ship, k);
                        02=if(temp_item == item)
                        03 ^{
                            01=check = 2;
                            02=break;
                        }
                    }
                }
            }
            //И под конец все прочие корабли в системе
            08=if(!check)
            09 ^{
                01=for(j = 0; j < StarShips(star); j = j + 1)
                02 ^{
                    01=ship = StarShips(star, j);
                    02=for(k = 0; k < ShipItems(ship); k = k + 1)
                    03 ^{
                        01=temp_item = ShipItems(ship, k);
                        02=if(temp_item == item)
                        03 ^{
                            01=if(ShipType(ship) == 'DefPlatform') check = 2;
                            01=else check = 3;
                            02=break;
                        }
                    }
                }
            }
            10=if(check != 2)
            11 ^{
                01=if(EqModule(item))
                02 ^{
                    01=int mod = EqModule(item);
                    02=EqModule(item, -1);
                    03=str name = Name(item);
                    04=EqModule(item, mod);
                }
                03=else name = Name(item);
                04=EqSpecial(item, -1);
                05=EqSpecial(item, FindBonusByCustomTag('AssemblyDroidNerfed'));
                06=SetName(item, name);
            }
        }
        05=else if(ScriptItemActionType(t_OnItemDeEquip))
        06 ^{
            01=item = CurItem();
            02=if(EqModule(item))
            03 ^{
                01=mod = EqModule(item);
                02=EqModule(item, -1);
                03=name = Name(item);
                04=EqModule(item, mod);
            }
            05=else name = Name(item);
            06=EqSpecial(item, -1);
            07=EqSpecial(item, FindBonusByCustomTag('AssemblyDroidNormal'));
            08=SetName(item, name);
        }
        09=else if(ScriptItemActionType(t_OnItemDestroy))
        10 ^{
            01=toAddAssemblyDroids = toAddAssemblyDroids + 1;
        }
    }
}
